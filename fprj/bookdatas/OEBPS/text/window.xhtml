	var obj,
		target = window,
		r ='<table class="objects">';

	r += '<col class="title" /><col class="type" /><col class="note" />';

	for (var obj in target ) {
		if( typeof target[obj] !== "function" ) {
			r += [
			"<tr><th>" + obj + "</th>",
			"<td>", typeof target[obj] , "</td>",
			"<td>", target[obj], "</td></tr>\n" ].join("");
		}
	}
	r +="</table>";
	document.getElementById("r2").innerHTML = r;
